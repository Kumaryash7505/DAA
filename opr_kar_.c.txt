#include <stdio.h>
int main()
{
    int A[100];
    int n, hit = 0;
    printf("Enter the number of processes\n");
    scanf("%d", &n);
    printf("Enter the incoming stream\n");
    for (int i = 0; i < n; i++)
        scanf("%d", &A[i]);
    int f;
    printf("Enter the number of frames\n");
    scanf("%d", &f);
    int count[100] ;
    int temp[100] ;
    for (int i = 0; i < f; i++)
        temp[i] = -1;


    for (int i = 0; i < n; i++)
    {
        int flag = 0;
        for (int j = 0; j < f; j++)
        {
            if (A[i] == temp[j])
            {
                flag = 1;
                hit++;
            }
            else if (temp[j] == -1)
            {
                flag = 1;
                temp[j] = A[i];
                break;
            }
        }
        if (flag == 0)
        {
            for (int a = 0; a < f; a++)
            {
                int flag1 = 0;
                for (int k = i; k < n; k++)
                {
                    if (temp[a] == A[k])
                    {
                        flag1 = 1;
                        count[a] = k;
                        break;
                    }
                }
                if (flag1 == 0)
                    count[a] = 10000;
            }
            int max = 0, ind;
            for (int m = 0; m < f; m++)
            {

                if (count[m] > max)
                {
                    max = count[m];
                    ind = m;
                }
            }
            temp[ind] = A[i];
        }
        printf("\n");
        for (int i = 0; i < f; i++)
        {
            if (temp[i] != -1)
                printf("%d ", temp[i]);
            else
                printf("-- ");
        }
    }

    printf("hit = %d", hit);
    printf("miss = %d",(n- hit));

}
