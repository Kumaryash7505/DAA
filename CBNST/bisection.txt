#include<stdio.h>
#include<math.h>
#include<conio.h>
#define e 0.0001
#define f(x) x*x -20

void main()
{
    float x0,x1,x2;
    float f0,f1,f2;
    int i=0;
    printf("enter the limit x0 and x1 : ");
    scanf("%f %f",&x0, &x1);
    do{
        f0=f(x0);
        f1=f(x1);
        x2=(x0+x1)/2.0;
        f2=f(x2);
        if(f0*f2<0){
        x1=x2;
        }
        else{
        x0=x2;
        }
        i++;
        printf("no of iteration %d",i);
        printf("\troot = %f", x2);
        printf("\tfunction %f\n", f2);
    }while(fabs(f2)>e);
}
