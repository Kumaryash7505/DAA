#include<stdio.h>
#include<math.h>
#include<conio.h>
#define e 0.0001
#define f(x) x*x -20
void main()
{
    int i=0;
    float x0,x1,x2;
    float f0,f1,f2;
    printf("enter the limits x0 and x1 :");
    scanf("%f%f",&x0,&x1);
    do{
    f0=f(x0);
    f1=f(x1);
    x2=x0 - ((x1-x0)*f0)/(f1-f0);
    f2=f(x2);
    if(f0*f2>0){
        x0=x2;
        f0=f2;
    }
    else{
        x1=x2;
        f1=f2;
    }
    i++;
    printf("no. of iteration : %d",i);
    printf("\t root : %f\t",x2);
    printf("function : %f\n",f2);
    }while(fabs(f2)>e);

}
