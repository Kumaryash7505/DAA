#include<stdio.h>
#include<math.h>
#define f(x,y) x+y
 void main()
 {
 int n,i;
 float x,xn,y,h,s,s1,s2,s3,s4;
 
 printf("enter the value of h,x0,y0: ");
 scanf("%f %f %f",&h,&x,&y);
 printf("enter the value of xn:");
 scanf("%f",&xn);
 while(x<xn)
 {
 s1=f(x,y);
 s2=f((x+(h/2)),(y+(h/2)*s1));
 s3=f((x+(h/2)),(y+(h/2)*s2));
 s4=f((x+h),(y+h*s3));
 s=(s1+(2*s2)+(2*s3)+s4)/6;
 y=y+(h*s);
 x=x+h;
 }
 printf("Output = xn:%f,y0:%f",x,y);
 }
