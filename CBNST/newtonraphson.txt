#include<stdio.h>
#include<math.h>
#include<conio.h>
#define e 0.0001
#define f(x) x*x*x -4*x +1
#define df(x) x*x -4
void main()
{
    float x0,x1,f0,f1,df0;
    int i=0;
    printf("enter the vale of x0 :");
    scanf("%f",&x0);
    do
    {
        f0=f(x0);
        df0=df(x0);
        x1=x0-(f0/df0);
        f1=f(x1);
        x0=x1;
        i++;
        printf("no. of iteration : %d",i);
        printf("\troot: %f\t",x1);
        printf("function :%f\n",f1);
    }while(fabs(f1)>e);
}
