#include<stdio.h>
int main()
{
    int i, j, n;
    float x[10], y[10], sum, u, p;
    printf("Enter the number of data: ");
    scanf("%d", &n);
    printf("Enter the data: ");
    for(i=0; i<n; i++)
    {
      scanf("%f %f", &x[i], &y[i]);
    }
    printf("Enter the value of u: ");
    scanf("%f", &u);
    sum = 0;
    for(i=0; i<n; i++)
    {
        p = 1;
        for(j=0; j<n; j++)
        {
            if(i!=j)
            {
                p = p * (u - x[j]) / (x[i] - x[j]);
            }
            
        }
        sum = sum + p * y[i];
    }
    printf("The value of f(%f) is %f", u, sum);
    return 0;
}
