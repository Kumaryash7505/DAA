#include<stdio.h>
#include<math.h>
#define f(x) 1/(4*x + 5)
int main()
{
   int n;
   printf("Enter the number of intervals: ");
   scanf("%d",&n);
   float a,b,h;
   printf("Enter the lower limit: ");
   scanf("%f",&a);
   printf("Enter the upper limit: ");
   scanf("%f",&b);
   h=(b-a)/n;
   float x[n+1],y[n+1];
    for(int i=0;i<=n;i++)
    {
        x[i]=a+i*h;
        y[i]=f(x[i]);
    }
    float sum=0;
    for(int i=1;i<n;i++)
    {
        if(i%2==0)
        {
            sum=sum+2*y[i];
        }
        else
        {
            sum=sum+4*y[i];
        }
    }
    sum=sum+y[0]+y[n];
    sum=sum*h/3;
    printf("The value of integral is %f",sum);
    return 0;
}
