#include<stdio.h>
#include<math.h>

int main()
{
  int n,inp,i,j;
  float p,q,r,fact;
  printf("Enter the total size of input:");
  scanf("%d",&n);

  printf("\n Enter the value of x :\n");
   float arr[n][n+1];
  for(i=0;i<n;i++)
  {
     scanf("%f",&arr[i][0]);
  }
  printf("\n enter the value of y :\n");

  for(i=0;i<n;i++)
  {
     scanf("%f",&arr[i][1]);
  }

   printf("Enter the value to find:");
     scanf("%d",&inp);
   for(j=2;j<n+1;j++)
    {
        for(i=0;i<n-j+1;i++)
           {
                arr[i][j] = arr[i+1][j-1]-arr[i][j-1];
           }
    }

    p= (inp - arr[0][0])/(arr[1][0]-arr[0][0]);
    r=arr[0][1];
    q=p;
    fact=1;
    for(i=2;i<=n;i++)
    {
        r=r+(q*arr[0][i])/fact;

        fact=fact*i;
        q=q*(p-(i-1));
    }
    printf("\n\nValue at X=%d is = %f", inp,r);
}
