#include<stdio.h>
#include<stdlib.h>

void main(){
int i, n  ; 
printf("enter number of process\n");
scanf("%d",&n);

int arrival[n+1] , burst[n+1], completion[n+1] , tat[n+1], waiting[n+1];
printf("enter arrival time \n");
for (int i = 0; i < n; i++)
{
    scanf("%d",&arrival[i]);
}

printf("enter burst time \n");
for (int i = 0; i < n; i++)
{
    scanf("%d",&burst[i]);
}


completion[0] = 0 ;

for (int i = 0; i < n; i++)
{
        completion[i+1] = burst[i] + completion[i] ;
}

if (arrival[0]!=0)
{
    for (int i = 0; i <= n; i++)
    {
        completion[i+1] +=arrival[0];
    }
    
}



for(int i = 0 ; i<n ; i++)
{
    tat[i] = completion[i+1] - arrival[i];
}

for(int i = 0 ; i<n ; i++)
{
    waiting[i] = tat[i] - burst[i];
}

float avg_wt = 0 , avg_tat= 0;
printf("process\tarrival\t\tburst\t\tcompletion\tTAT\t\twaiting \n");
for(int i =0 ; i < n ; i++)
{
    printf("%d\t %d\t\t %d \t\t %d\t\t %d \t\t %d \n",i+1,arrival[i],burst[i],completion[i+1],tat[i],waiting[i]);
    avg_tat+=tat[i];
    avg_wt+=waiting[i];
}

avg_tat = avg_tat/n;
avg_wt = avg_wt/n;

printf("average waiting time = %.2f\n", avg_wt);
printf("average turn around time  =%.2f",avg_tat);


}
